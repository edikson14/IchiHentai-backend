const mongoose = require('mongoose');
const Schema = mongoose.Schema;

// Ini adalah struktur atau "Cetak Biru" untuk setiap dokumen manga di database
const ChapterSchema = new Schema({
    id: Number,
    title: String,
    pages: [String],
    releaseDate: Date
});

const MangaSchema = new Schema({
    title: { type: String, required: true },
    lang: String,
    price: Number,
    author: String,
    status: String,
    cover: String, // Akan menyimpan gambar dalam format Base64
    synopsis: String,
    genres: [String],
    views: { type: Number, default: 0 },
    likes: { type: Number, default: 0 },
    chapters: [ChapterSchema],
    comments: [{
        user: String,
        text: String,
        timestamp: { type: Date, default: Date.now }
    }]
}, {
    // Memberi stempel waktu otomatis kapan data dibuat/diubah
    timestamps: true 
});

module.exports = mongoose.model('Manga', MangaSchema);
